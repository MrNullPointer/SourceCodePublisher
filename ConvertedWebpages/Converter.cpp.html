#include "Converter.h"

#ifdef TEST_CONVERTER

int main(int argc, char* argv[]) 
{
	Title("Demonstrate Converter package");
	ProcessCmdLine pcl(argc, argv);
	preface("Command Line: ");
	pcl.showCmdLine();
	if (pcl.parseError()) {
		pcl.usage();
		std::cout << "\n\n";
		return 1;
	}
	bool hasS = false;
	if (pcl.hasOption('s')) {
		hasS = true;
	}
	std::string outputpath = Path::getFullFileSpec("../convertedPages/");
	
//1. search filterfiles

	SearchFiles searchfiles(pcl.directoryPath(), pcl.patterns(), pcl.regexes(), hasS);
	std::vector<std::string> filterfiles = searchfiles.doSearching();
	
//2. use filterfiles and directoryPath outputpath to do convert and then generate html pages

	CodeConvert converter(pcl.directoryPath(), filterfiles, outputpath);
	converter.doConversion();
	return 0;
}

#endif 
// TEST_CONVERTER
